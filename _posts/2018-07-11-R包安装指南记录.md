pkg.list = c("ggplot2","ggthemes", "smooth", "cluster", "pheatmap",  "dplyr", "data.table", "Hmisc")
myInstallPackages <- function(pkgs) {
    new.pkg <- pkgs[!(pkgs %in% installed.packages()[, "Package"])]
    if (length(new.pkg)) {
         install.packages(new.pkg, dependencies=TRUE, repos="https://mirrors.tongji.edu.cn/CRAN/")  #, destdir="/nethome/whh/.local/R-3.5.1/R/library"
    }
}
myInstallPackages(pkg.list)
